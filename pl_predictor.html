from flask import Flask, request, jsonify
from flask_cors import CORS
from pyngrok import ngrok
import threading, pandas as pd, numpy as np, joblib

# Set your ngrok authtoken (only needed once per session)
ngrok.set_auth_token("PASTE_YOUR_NGROK_TOKEN_HERE")

# Load your saved model
model = joblib.load("final_random_forest_model.pkl")
FEATURE_COLS = [col for col in X.columns if col != 'Season']

app = Flask('__main__')
CORS(app)

@app.route('/predict', methods=['POST'])
def predict_endpoint():
    data    = request.json
    team    = data['team_name']
    mp      = int(data['matches_played'])
    wins    = int(data['wins'])
    draws   = int(data['draws'])
    losses  = int(data['losses'])
    gf      = int(data['goals_for'])
    ga      = int(data['goals_against'])
    sot     = float(data['sot_per_match'])
    sn      = int(data['season_num'])

    row = pd.Series(0, index=FEATURE_COLS, dtype=object)
    row['MatchesPlayed'] = 38
    row['W']            = wins
    row['D']            = draws
    row['L']            = losses
    row['GF']           = gf
    row['GA']           = ga
    row['GD']           = gf - ga
    row['SOT_per_match']= float(sot)
    row['Season_num']   = sn
    team_col = f"Team_{team}"
    if team_col in row.index:
        row[team_col] = 1

    input_df = pd.DataFrame([row])
    for c in input_df.columns:
        if c.startswith('Team_'):
            input_df[c] = input_df[c].astype(int)

    pts_full = model.predict(input_df)[0]
    scaled   = (pts_full / 38) * mp
    return jsonify({'predicted_points': round(scaled, 2)})

public_url = ngrok.connect(5001).public_url
print(f"✅ Paste this URL into the GUI: {public_url}")
threading.Thread(target=lambda: app.run(port=5001)).start()
